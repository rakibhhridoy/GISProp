\chapter{Risks \& Mitigation}

\section{Risk 1: Competitive Pressure}

\textbf{Risk:} Competitors add ML/apps, reducing our differentiation

\textbf{Probability:} Medium (40-50\%)

\textbf{Impact:} Reduced pricing power, slower adoption

\textbf{Mitigation:}
\begin{itemize}
    \item Speed: Execute faster than competitors can respond
    \item Lock-in: Deep NGO relationships with exclusive agreements
    \item Data Moat: 18-24 months of farm data competitors can't replicate
    \item Focus: Compete on farmers (different economics than government projects)
\end{itemize}

\section{Risk 2: NGO Partnership Challenges}

\textbf{Risk:} NGO partners uninterested or execute poorly

\textbf{Probability:} Medium (30\%)

\textbf{Impact:} Farmer app revenue takes 2+ years longer to scale

\textbf{Mitigation:}
\begin{itemize}
    \item Proof of concept: Pilot with one NGO (5K farmers) before full scale
    \item Incentive alignment: Commission structure that rewards NGO results
    \item Backup channels: Direct sales + government schools as alternatives
    \item Multiple partners: Partner with 3-5 NGOs, not dependent on one
\end{itemize}

\section{Risk 3: Regulatory Changes}

\textbf{Risk:} CAAB tightens regulations or adds new restrictions

\textbf{Probability:} Low-Medium (20-30\%)

\textbf{Impact:} Could increase costs 20-50\%, delay operations

\textbf{Mitigation:}
\begin{itemize}
    \item Engagement: Build relationships with CAAB officials
    \item Advocacy: Participate in regulatory discussions
    \item Flexibility: Build model that works under multiple categories
    \item Contingency: Plan regional expansion to other countries if needed
\end{itemize}

\section{Risk 4: Technology Integration}

\textbf{Risk:} LiDAR + Data Science integration proves harder than expected, models underperform

\textbf{Probability:} Low (15-20\%)

\textbf{Impact:} Farmer apps don't deliver promised ROI, adoption stalls

\textbf{Mitigation:}
\begin{itemize}
    \item Validation: Test models on pilot farms in Year 1
    \item Incremental: Start with simple models, advance to complex
    \item Expertise: Hire experienced ML/GIS engineers
    \item Fallback: Apps work with traditional indices if LiDAR integration incomplete
\end{itemize}

\section{Risk 5: Talent Constraints}

\textbf{Risk:} Difficulty recruiting experienced pilots, Data Science engineers, GIS specialists

\textbf{Probability:} High (60\%)

\textbf{Impact:} Hiring delays slow operations 3-6 months

\textbf{Mitigation:}
\begin{itemize}
    \item Poaching: Recruit from competitors (they have market knowledge)
    \item Training: Partner with Avion Aerospace for pilot pipeline
    \item Remote: Hire Data Science engineers from South Asia with competitive salaries
    \item Equity: Offer stock options to attract risk-takers
    \item Universities: Partner with BAU for early-career talent
\end{itemize}

\section{Risk 6: Farmer Adoption Delays}

\textbf{Risk:} Farmers reluctant to adopt apps or won't pay

\textbf{Probability:} Low-Medium (25-35\%)

\textbf{Impact:} App revenue slower than projected

\textbf{Mitigation:}
\begin{itemize}
    \item Proof of demand: 127K farmers already using drones (World Bank data)
    \item Distribution: NGO partners have farmer trust (40+ year relationships)
    \item Pricing: BDT 5-10K/year is 0.3-1\% of farmer income with clear ROI
    \item Freemium: Offer free version with limited features if needed
    \item Stories: Collect and share real farmer success stories
\end{itemize}
